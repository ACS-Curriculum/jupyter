Day1: Variables1


function setup() {
  createCanvas(400, 400); // Create a 400x400 pixel canvas
}

function draw() {
  background(220); // Set background color to light gray

  // Set fill color to yellow for the face
  fill(255, 255, 0);

  // Draw face (head) - a large yellow circle
  ellipse(200, 200, 150, 150);

  // Draw eyes - two white circles
  fill(255); // Set fill color to white for the eyes
  ellipse(170, 170, 30, 30); // Left eye
  ellipse(230, 170, 30, 30); // Right eye

  // Draw eyeballs - two black circles
  fill(0); // Set fill color to black for the pupils
  ellipse(180, 170, 10, 10); // Left pupil
  ellipse(240, 170, 10, 10); // Right pupil

  // Draw mouth (default happy face)
  noFill(); // No fill for the mouth (outline only)
  arc(200, 220, 100, 50, 0, PI); // Draw an arc for a smiling mouth
}

Day1: Variables2


let faceX = 200, faceY = 200, faceSize = 150; // Face position and size
let eyeOffsetX = 30, eyeOffsetY = 30, eyeSize = 50, pupilSize = 30; // Eye properties
let mouthWidth = 100, mouthHeight = 50, mouthY = 220; // Mouth properties

function setup() {
  createCanvas(400, 400); // Create a 400x400 pixel canvas
}

function draw() {
  background(220); // Set background color to light gray

  // Set fill color to yellow for the face
  fill(255, 255, 0);
  ellipse(faceX, faceY, faceSize, faceSize); // Draw face

  // Set fill color to white for the eyes
  fill(255);
  ellipse(faceX - eyeOffsetX, faceY - eyeOffsetY, eyeSize, eyeSize); // Left eye
  ellipse(faceX + eyeOffsetX, faceY - eyeOffsetY, eyeSize, eyeSize); // Right eye

  // Set fill color to black for the pupils
  fill(0);
  ellipse(faceX - eyeOffsetX + 10, faceY - eyeOffsetY, pupilSize, pupilSize); // Left pupil
  ellipse(faceX + eyeOffsetX + 10, faceY - eyeOffsetY, pupilSize, pupilSize); // Right pupil

  // Draw mouth (default happy face)
  noFill();
  arc(faceX, mouthY, mouthWidth, mouthHeight, 0, PI);
}
