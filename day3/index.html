function setup() {
  createCanvas(600, 400); // Larger canvas to fit multiple faces
}
function draw() {
  background(220);
  let startX = 100; // Starting x position
  let startY = 200; // Y position for all faces
  let faceSize = 100; // Face size
  let spacing = 120; // Space between faces

  for (let i = 0; i < 4; i++) { 
    drawFace(startX + i * spacing, startY, faceSize); // Draw faces in a row
  }
}
// Function to draw a face with customizable position and size
function drawFace(x, y, size) {
  let eyeOffsetX = size * 0.2;
  let eyeOffsetY = size * 0.2;
  let eyeSize = size * 0.2;
  let pupilSize = size * 0.07;
  let mouthWidth = size * 0.67;
  let mouthHeight = size * 0.33;
  let mouthY = y + size * 0.15;
  // Draw face
  fill(255, 255, 0);
  ellipse(x, y, size, size);
  // Draw eyes
  fill(255);
  ellipse(x - eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize);
  ellipse(x + eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize);

  // Draw pupils
  fill(0);
  ellipse(x - eyeOffsetX + eyeSize * 0.3, y - eyeOffsetY, pupilSize, pupilSize);
  ellipse(x + eyeOffsetX + eyeSize * 0.3, y - eyeOffsetY, pupilSize, pupilSize);

  // Draw mouth
  noFill();
  stroke(0);
  strokeWeight(2);
  arc(x, mouthY, mouthWidth, mouthHeight, 0, PI);
}

day3.4


function setup() {
  createCanvas(600, 400); // Canvas for multiple faces
}

function draw() {
  background(220);
  
  let startX = 100; // Starting x position
  let startY = 100; // Starting y position
  let faceSize = 100; // Face size
  let spacingX = 120; // Horizontal spacing
  let spacingY = 120; // Vertical spacing
  let cols = 4; // Number of faces per row
  let rows = 3; // Number of rows

  for (let j = 0; j < rows; j++) { // Loop for rows
    for (let i = 0; i < cols; i++) { // Loop for columns
      push(); // Save transformation state
      translate(startX + i * spacingX, startY + j * spacingY); // Move to grid position
      drawFace(0, 0, faceSize); // Draw face at (0,0) in translated space
      pop(); // Restore transformation state
    }
  }
}

// Function to draw a face with customizable position and size
function drawFace(x, y, size) {
  let eyeOffsetX = size * 0.2;
  let eyeOffsetY = size * 0.2;
  let eyeSize = size * 0.2;
  let pupilSize = size * 0.07;
  let mouthWidth = size * 0.67;
  let mouthHeight = size * 0.33;
  let mouthY = y + size * 0.15;

  // Draw face
  fill(255, 255, 0);
  ellipse(x, y, size, size);

  // Draw eyes
  fill(255);
  ellipse(x - eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize);
  ellipse(x + eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize);

  // Draw pupils
  fill(0);
  ellipse(x - eyeOffsetX + eyeSize * 0.3, y - eyeOffsetY, pupilSize, pupilSize);
  ellipse(x + eyeOffsetX + eyeSize * 0.3, y - eyeOffsetY, pupilSize, pupilSize);

}
Note that the inner loop in a "nested loop" works harder than the outside loop. The outside loop runs once, then drops into the inner loop. The inner loop then runs all the way through before going back up to the outer loop that again runs once before dropping down to run all the way through, etc.

Day3.5


let r = 0;
let g = 0;
let b = 0;

function setup() {
  createCanvas(400, 400);
}

function draw() {
  background(30);

  let cx = width / 2;
  let cy = height / 2;
  let radius = 100;

  beginShape();
  for (let i = 0; i < 10; i++) {
    let angle = (TWO_PI * i) / 10;
    let x = cx + cos(angle) * radius;
    let y = cy + sin(angle) * radius;

    if (frameCount % 180 < 90) {
      // Wobbles for 1.5 sec, stable for 1.5 sec
      x += random(-10, 10);
      y += random(-10, 10);
      r = random(0, 255);
      g = random(0, 255);
      b = random(0, 255);
    }
    noFill();
    stroke(r, g, b);
    strokeWeight(6);
    vertex(x, y);
  }
  endShape(CLOSE);
}
