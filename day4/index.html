function setup() {
  createCanvas(600, 400); // Larger canvas to fit multiple faces
}
function draw() {
  background(220);
  let startY = 200; // Y position for all faces
  let faceSize = 100; // Face size
  let spacing = 120; // Space between faces
  
  // Array to store x-positions for faces
  let faceXPositions = [100, 220, 340, 460]; // x-positions of the faces
  for (let i = 0; i < faceXPositions.length; i++) {
    push(); // Save current transformation state
    translate(faceXPositions[i], startY); // Translate to the correct x-position
    drawFace(0, 0, faceSize); // Draw face at translated position
    pop(); // Restore previous transformation state
  }
}
// Function to draw a face with customizable position and size
function drawFace(x, y, size) {
  let eyeOffsetX = size * 0.2;
  let eyeOffsetY = size * 0.2;
  let eyeSize = size * 0.2;
  let pupilSize = size * 0.07;
  let mouthWidth = size * 0.67;
  let mouthHeight = size * 0.33;
  let mouthY = y + size * 0.15;
  // Draw face
  fill(255, 255, 0);
  ellipse(x, y, size, size);
  // Draw eyes
  fill(255);
  ellipse(x - eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize);
  ellipse(x + eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize);
  // Draw pupils
  fill(0);
  ellipse(x - eyeOffsetX + eyeSize * 0.3, y - eyeOffsetY, pupilSize, pupilSize);
  ellipse(x + eyeOffsetX + eyeSize * 0.3, y - eyeOffsetY, pupilSize, pupilSize);
  // Draw mouth
  noFill();
  stroke(0);
  strokeWeight(2);
  arc(x, mouthY, mouthWidth, mouthHeight, 0, PI);
}
}

//declare an empty array variable
var myCollection = [];

function setup(){
  createCanvas(400,400);
  // you can fill the array by referencing the array's
  // indices
  for(var i =1; i <= 5; i++){
    myCollection[i-1] = random(1,5);
  }
  
  //or you can push items onto the array
  myCollection.push("hello");
  myCollection.push("world");
  myCollection.push(3);
  myCollection.push(2000.2);
  
  text(myCollection,0,40);

  //how large is our array?
  text(myCollection.length, 10,height-30);
}

function setup() {
  createCanvas(400,400);

  var evenNumbers = [];
  for(var i=0; i < 100; i++){
    evenNumbers[i] = i *2;
  }
  text(evenNumbers,0,0);
  
  var tenths = [];
  var number = 0.0;
  for(var j=0; j < 10; j++){
    tenths[j] = (number += 0.1);
  }

Arrays drawing


var xPos = [];
var yPos =[];
var arraySize = 10;
function setup() {
  createCanvas(400,400);
  for(var i=0; i < arraySize; i++){
    xPos[i] = random(500.0);
    yPos[i] = random(500.0);
  }
}

function draw() {
  background(255);
  for(var i=0; i < xPos.length; i++){
    fill(255,0,0,50);//r, g, b, a
    rect(xPos[i], yPos[i], 300, 250);
  }
}
